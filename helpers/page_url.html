Page URL Helper

Usage example: {{page_url 2}}

Returns the URL for the page specified in the current object context.

We use the name page_url to match the helper for consistency: jscs:disable requireCamelCaseOrUpperCaseIdentifiers

import SettingsStore from '../stores/SettingsStore'; var page_url, pageUrl; page_url = function (context, block) { /*jshint unused:false*/ var url = SettingsStore.getSiteUrl(); if (this.tagSlug !== undefined) { url += '/tag/' + this.tagSlug; } if (this.authorSlug !== undefined) { url += '/author/' + this.authorSlug; } if (context > 1) { url += '/page/' + context; } url += '/'; return url; };
