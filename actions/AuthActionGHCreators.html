import HpConstants from '../constants/HpConstants'; const HpDispatcher = require('../dispatchers/HpDispatcher'); const ActionTypes = HpConstants.ActionTypes;

AuthActionGHCreators

class AuthActionGHCreators {

Method receiveAuthentication

Receive authentication information from GitHub

Parameters:

  • values must be an Object.

    • values.authentication must be an Object.
      (Email or Username of the user)

    • values.authentication.permissions must be an Object.

    • values.authentication.permissions.admin must be a boolean.
      (true if user is admin)

    • values.authentication.permissions.push must be a boolean.
      (true if user can push)

    • values.authentication.permissions.put must be a boolean.
      (true if user can put)

    • values.token must be a String.
      (Token to use to query GitHub API)

    • values.message must be an Object.

    • values.message.content must be a String.
      (Content of the message to show)

    • values.message.title must be a String.
      (Title of the message to shoz)

    • values.message.type must be a String.
      (Type of the message)

receiveAuthentication(values) { console.info('AuthActionGHCreators - receiveAuthentication'); console.log('AuthActionGHCreators - receiveAuthentication', values); HpDispatcher.handleViewAction({ type: ActionTypes.RECEIVE_AUTHENTICATION, authentication: values.authentication, message: values.message }); } } export default new AuthActionGHCreators();
